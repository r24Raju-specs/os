import threading
import random

buffer=[]
max_size = 5

empty = threading.Semaphore(max_size)
full = threading.Semaphore(0)
mutex = threading.Lock()

def producer():
    for _ in range(10):
        item= random.randint(1,100)
        empty.acquire()
        mutex.acquire()

        buffer.append(item)
        print("Producer:",item)

        mutex.release()
        full.release()

def consumer():
    for _ in range(10):
        full.acquire()
        mutex.acquire()

        item=buffer.pop()
        print("Consumer:",item)

        mutex.release()
        empty.release()

t1 = threading.Thread(target=producer)
t2 = threading.Thread(target=consumer)

t1.start()
t2.start()

t1.join()
t2.join()
